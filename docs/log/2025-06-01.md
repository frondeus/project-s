```
(let :left {
    (let :x 4)
    :another x
    :key {
      :a 1
      :b (+ 1 (root :another))
    }
  })

(let :right {
  :another 9
  
  (if (has? super :key)
    '(:key 
      (+ (super :key) {
        :c 3
      })
    )
    '(:key {:c 3})
  )

})

(+ left right)
```
